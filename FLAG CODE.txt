<script>

const stripeControls = document.getElementById("stripeControls");

for (let i = 0; i < 13; i++) {
  const input = document.createElement("input");
  input.type = "color";
  input.value = i % 2 === 0 ? "#B22234" : "#FFFFFF";
  input.id = "stripe" + i;
  stripeControls.appendChild(input);
}

let unionImg = null;

document.getElementById("unionImage").addEventListener("change", function(e) {
  const file = e.target.files[0];
  const reader = new FileReader();

  reader.onload = function(event) {
    unionImg = new Image();
    unionImg.src = event.target.result;
    unionImg.onload = drawFlag;
  };

  if (file) reader.readAsDataURL(file);
});

function drawStars(ctx, width, height) {
  const starColor = document.getElementById("starColor").value;
  ctx.fillStyle = starColor;

  for (let y = 10; y < height; y += 25) {
    for (let x = 10; x < width; x += 25) {
      ctx.beginPath();
      ctx.arc(x, y, 3, 0, Math.PI * 2);
      ctx.fill();
    }
  }
}

function drawFlag() {
  const canvas = document.getElementById("canvas");
  const ctx = canvas.getContext("2d");

  const stripeHeight = canvas.height / 13;
  const texture = document.getElementById("texture").value;
  const showStars = document.getElementById("starsToggle").checked;

  // STRIPES
  for (let i = 0; i < 13; i++) {
    const color = document.getElementById("stripe" + i).value;
    ctx.fillStyle = color;
    ctx.fillRect(0, i * stripeHeight, canvas.width, stripeHeight);
  }

  const unionWidth = canvas.width * 0.4;
  const unionHeight = stripeHeight * 7;

  // UNION BASE
  const unionColor = document.getElementById("unionColor").value;
  ctx.fillStyle = unionColor;
  ctx.fillRect(0, 0, unionWidth, unionHeight);

  // TEXTURE
  if (texture === "wood") {
    ctx.fillStyle = "rgba(139,69,19,0.3)";
    ctx.fillRect(0, 0, canvas.width, canvas.height);
  }

  if (texture === "burnt") {
    ctx.fillStyle = "rgba(0,0,0,0.5)";
    ctx.fillRect(0, 0, canvas.width, canvas.height);
  }

  // UNION IMAGE
  if (unionImg) {
    ctx.drawImage(unionImg, 0, 0, unionWidth, unionHeight);
  }

  // STARS
  if (showStars && !unionImg) {
    drawStars(ctx, unionWidth, unionHeight);
  }
}

function download() {
  const canvas = document.getElementById("canvas");
  const link = document.createElement("a");
  link.download = "custom_flag.png";
  link.href = canvas.toDataURL();
  link.click();
}

drawFlag();
